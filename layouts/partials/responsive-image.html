{{/* Responsive Image Partial Automatically generates responsive images in multiple sizes and WebP format. Falls back gracefully to original image if processing is not possible. URL Strategy: - Uses .RelPermalink for processed image resources (includes fingerprint hash in production) - Uses relURL for unprocessable images (external URLs, unsupported formats) - All URLs are relative to support subdirectory deployments and baseURL configuration Usage: {{ partial "responsive-image" (dict "src" "/images/example.jpg" "alt" "Image description" "class" "my-class" "sizes" "(max-width: 768px) 100vw, 50vw" "width" 800 "height" 600 "loading" "lazy" "context" . ) }} Parameters: - src: Image path (required) - can be relative path from assets or absolute URL - alt: Alt text for accessibility (required) - class: CSS classes to add to img element (optional) - sizes: Sizes attribute for responsive images (optional, default: "100vw") - width: Desired max width (optional, used for single image resize) - height: Desired max height (optional) - loading: Loading attribute - "lazy" or "eager" (optional, default: "lazy") - context: Page context for resource access (optional, default: site) - linkUrl: If provided, wraps image in anchor link (optional) - linkClass: CSS class for anchor wrapper (optional) - figcaption: Caption text for figure element (optional) */}} {{- $src := .src -}} {{- $alt := .alt | default "" -}} {{- $class := .class | default "" -}} {{- $sizes := .sizes | default "100vw" -}} {{- $maxWidth := .width | default 1280 -}} {{- $maxHeight := .height -}} {{- $loading := .loading | default "lazy" -}} {{- $context := .context -}} {{- $linkUrl := .linkUrl -}} {{- $linkClass := .linkClass | default "" -}} {{- $figcaption := .figcaption -}} {{/* Responsive image widths from site config or defaults */}} {{- $responsiveWidths := site.Params.images.sizes | default (slice 320 480 768 1024 1280) -}} {{/* Initialize variables */}} {{- $image := false -}} {{- $isProcessable := false -}} {{- $originalSrc := $src -}} {{/* Try to get image as a resource for processing */}} {{- if $src -}} {{/* Remove leading slash for assets lookup */}} {{- $assetPath := strings.TrimPrefix "/" $src -}} {{/* Try global assets first */}} {{- with resources.Get $assetPath -}} {{- $image = . -}} {{- $isProcessable = true -}} {{- end -}} {{/* Try page resources if context is a Page with Resources */}} {{/* Only check for page resources if image not found in global assets */}} {{/* Skip if context is a map/dict (data file) or site wrapper */}} {{- if not $image -}} {{- $isPage := false -}} {{- if $context -}} {{/* Check if context has a Kind (indicating it's a Page) */}} {{- $kind := "" -}} {{- with $context.Kind -}} {{- $kind = . -}} {{- $isPage = true -}} {{- end -}} {{- end -}} {{- if $isPage -}} {{- with $context.Resources -}} {{- with .GetMatch (path.Base $src) -}} {{- $image = . -}} {{- $isProcessable = true -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{/* Build the image output */}} {{- if and $image $isProcessable -}} {{/* Image is processable - generate responsive versions */}} {{/* Determine format support */}} {{- $supportsWebP := true -}} {{- $format := $image.MediaType.SubType -}} {{/* Skip processing for SVG, GIF (animated), and already small images */}} {{- $skipProcessing := or (eq $format "svg+xml") (eq $format "svg") (eq $format "gif") -}} {{- if $skipProcessing -}} {{/* SVG/GIF - use original without processing */}} {{- else -}} {{/* Process image - generate srcset for responsive images */}} {{/* Generate WebP and original format versions */}} {{- $srcsetWebP := slice -}} {{- $srcsetOriginal := slice -}} {{- $fallbackImage := $image -}} {{/* Calculate aspect ratio for consistent cropping */}} {{- $aspectRatio := 0.0 -}} {{- if and $image.Width $image.Height -}} {{- $aspectRatio = div (float $image.Height) (float $image.Width) -}} {{- end -}} {{/* Generate images at each responsive width */}} {{- range $width := $responsiveWidths -}} {{- if le $width $image.Width -}} {{/* Resize to width, maintaining aspect ratio */}} {{- $resized := $image.Resize (printf "%dx" $width) -}} {{- $srcsetOriginal = $srcsetOriginal | append (printf "%s %dw" $resized.RelPermalink $width) -}} {{/* Generate WebP version */}} {{- if $supportsWebP -}} {{- $webp := $image.Resize (printf "%dx webp" $width) -}} {{- $srcsetWebP = $srcsetWebP | append (printf "%s %dw" $webp.RelPermalink $width) -}} {{- end -}} {{/* Use middle size as fallback */}} {{- if and (ge $width 768) (le $width 1024) -}} {{- $fallbackImage = $resized -}} {{- end -}} {{- end -}} {{- end -}} {{/* If image is smaller than smallest responsive width, use original */}} {{- if eq (len $srcsetOriginal) 0 -}} {{- $srcsetOriginal = $srcsetOriginal | append (printf "%s %dw" $image.RelPermalink $image.Width) -}} {{- if $supportsWebP -}} {{- $webp := $image.Resize (printf "%dx webp" $image.Width) -}} {{- $srcsetWebP = $srcsetWebP | append (printf "%s %dw" $webp.RelPermalink $image.Width) -}} {{- end -}} {{- $fallbackImage = $image -}} {{- end -}} {{/* Generate blur placeholder for loading state */}} {{- $placeholder := "" -}} {{- $blurredImage := $image.Resize "20x q20" -}} {{- if $blurredImage -}} {{- $placeholder = $blurredImage.RelPermalink -}} {{- end -}} {{/* Output picture element with WebP and fallback */}} {{- if $figcaption -}} {{- end -}} {{- end -}} {{- else -}} {{/* Image is not processable - use original src with lazy loading and error handling */}} {{- if $figcaption -}} {{- end -}} {{- end -}}
