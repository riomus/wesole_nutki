{{- /* BaseURL Path Helper Converts internal URLs to use proper basepath configuration by applying relURL. */ -}} {{- $url := . -}} {{- $result := $url -}} {{- if and $url (not (or (hasPrefix $url "http") (hasPrefix $url "#") (hasPrefix $url "mailto:") (hasPrefix $url "tel:"))) -}} {{- if not (hasPrefix $url "/") -}} {{- $url = printf "/%s" $url -}} {{- end -}} {{- $result = $url | relURL -}} {{- /* Robust fallback: check if base path is missing despite relURL */ -}} {{- $basePath := "" -}} {{- if site.BaseURL -}} {{- $basePath = urls.Parse site.BaseURL -}} {{- $basePath = $basePath.Path -}} {{- end -}} {{- if and $basePath (ne $basePath "/") -}} {{- $cleanBase := strings.TrimSuffix "/" $basePath -}} {{- if not (hasPrefix $result $cleanBase) -}} {{- /* Ensure we don't double slash */ -}} {{- $result = printf "%s%s" $cleanBase $result -}} {{- end -}} {{- end -}} {{- end -}} {{- $result -}}
