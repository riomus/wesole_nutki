{{- /* Image URL Helper Processes image URLs to ensure they work correctly with basepath configuration. */ -}} {{- $imagePath := . -}} {{- $result := $imagePath -}} {{- if $imagePath -}} {{- if or (hasPrefix $imagePath "http") (hasPrefix $imagePath "//") (hasPrefix $imagePath "data:") -}} {{- $result = $imagePath -}} {{- else -}} {{- if not (hasPrefix $imagePath "/") -}} {{- $imagePath = printf "/%s" $imagePath -}} {{- end -}} {{- $result = $imagePath | relURL -}} {{- /* Robust fallback: check if base path is missing despite relURL */ -}} {{- $basePath := "" -}} {{- if site.BaseURL -}} {{- $basePath = urls.Parse site.BaseURL -}} {{- $basePath = $basePath.Path -}} {{- end -}} {{- if and $basePath (ne $basePath "/") -}} {{- $cleanBase := strings.TrimSuffix "/" $basePath -}} {{- if not (hasPrefix $result $cleanBase) -}} {{- $result = printf "%s%s" $cleanBase $result -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- $result -}}
