# Automated Lighthouse and WCAG Audits Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add strict local Lighthouse and WCAG 2.2 A/AA checks that target perfect automated results, plus non-blocking GitHub Actions execution with retained reports.

**Architecture:** A JSON route manifest is the single source of truth for the bilingual audit surface. Lighthouse CI and a dedicated Axe/Playwright suite consume that manifest independently, while a separate GitHub workflow executes both checks with non-blocking audit steps and always uploads evidence. A Node contract test locks the configuration behavior before dependencies or implementation are added.

**Tech Stack:** Hugo, Node.js 20, Lighthouse CI, Playwright Chromium, Axe Core, GitHub Actions, Node test runner

---

- [ ] **Step 5: Add the WCAG 2.2 A/AA Axe suite**

Create `tests/audits/wcag.spec.ts`:

```ts
import AxeBuilder from '@axe-core/playwright';
import { expect, test } from '@playwright/test';
import { readFileSync } from 'node:fs';
import path from 'node:path';

const routes: string[] = JSON.parse(
  readFileSync(path.join(process.cwd(), 'audit/routes.json'), 'utf8'),
);

for (const route of routes) {
  test(`${route} has no automated WCAG A/AA violations`, async ({ page }, testInfo) => {
    await page.goto(route, { waitUntil: 'networkidle' });
    await page.evaluate(() => document.fonts.ready);

    const results = await new AxeBuilder({ page })
      .withTags(['wcag2a', 'wcag2aa', 'wcag21a', 'wcag21aa', 'wcag22aa'])
      .analyze();

    await testInfo.attach('axe-results', {
      body: JSON.stringify(results, null, 2),
      contentType: 'application/json',
    });

    expect(results.violations).toEqual([]);
  });
}
```

- [ ] **Step 1: Add the independent audit workflow**

Create `.github/workflows/quality-audit.yml` with checkout, Node 20/npm cache, Hugo `0.140.2`, `npm ci`, Playwright Chromium dependencies, non-blocking Lighthouse and Axe execution, always-on artifact upload, and an outcome summary:

```yaml
name: Quality audits

on:
  pull_request:
  push:
    branches: [master, main]
  workflow_dispatch:

permissions:
  contents: read

jobs:
  audit:
    runs-on: ubuntu-latest
    env:
      HUGO_VERSION: 0.140.2
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: npm

      - name: Install Hugo CLI
        run: |
          wget -O "${{ runner.temp }}/hugo.deb" "https://github.com/gohugoio/hugo/releases/download/v${HUGO_VERSION}/hugo_extended_${HUGO_VERSION}_linux-amd64.deb"
          sudo dpkg -i "${{ runner.temp }}/hugo.deb"

      - name: Install Node.js dependencies
        run: npm ci

      - name: Install Playwright Chromium
        run: npx playwright install --with-deps chromium

      - name: Run Lighthouse audit
        id: lighthouse
        continue-on-error: true
        run: npm run audit:lighthouse

      - name: Upload Lighthouse reports
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: lighthouse-reports
          path: test-results/lighthouse
          if-no-files-found: warn

      - name: Run WCAG audit
        id: wcag
        continue-on-error: true
        run: npm run audit:wcag

      - name: Upload WCAG reports
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: wcag-reports
          path: |
            test-results/wcag
            test-results/wcag-report
          if-no-files-found: warn

      - name: Summarize audit outcomes
        if: always()
        run: |
          echo "## Quality audits" >> "$GITHUB_STEP_SUMMARY"
          echo "- Lighthouse: ${{ steps.lighthouse.outcome }}" >> "$GITHUB_STEP_SUMMARY"
          echo "- WCAG: ${{ steps.wcag.outcome }}" >> "$GITHUB_STEP_SUMMARY"
          echo "Reports are attached as workflow artifacts." >> "$GITHUB_STEP_SUMMARY"
```
